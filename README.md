```bash
npm install --save use-url-state
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'use-url-state'
import 'use-url-state/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
